## Interval slider

<Story name="Interval slider" parameters={{ docs: { disable: true } }}>
  {() => {
    const [dates, setDates] = useState([]);
    const handleChange = array => {
      setDates(array);
      console.log(dates);
    };
    const dateRange = [
      new Date('2019-01-01'),
      new Date('2019-01-02'),
      new Date('2019-01-03'),
      new Date('2019-01-04'),
      new Date('2019-01-05'),
    ];
    return (
      <div style={{ width: '75%', marginTop: '5vh' }}>
        <DateSlider
          range={dateRange}
          values={dates}
          onChange={handleChange}
        />
      </div>
    );
  }}
</Story>
